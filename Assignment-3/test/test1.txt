
int add(int a, int b) {
    return a + b;
}

int factorial(int n) {
    if (n <= 1){ return 1;}
    else {return n * factorial(n - 1);}
    return 0;
}

int array_sum(int* arr, int size) {
    int sum = 0;
    for (int i = 0; i < size; i++) {
        sum = sum + arr[i];
    }
    return sum;
}

class Point {
    public:
    int x;
    int y;
};

double point_sum(Point p) {
    return (double)(p.x/p.y);
}

int main() {
     Point p;
    p.x = 3;
    p.y = 4;
    double p_sum = point_sum(p);
    int a = 5, b = 10;

    int result = add(a, b);
    if (result != 15) {
        printf("Arithmetic test failed: %d\n", result);
        return 1;
    }
    int num = 5;
    int fact = factorial(num);
    if (fact != 120) {
        printf("Recursion test failed: %d\n", fact);
        return 1;
    }
    printf("factorial of %d is %d\n", num, fact);

    int arr[5] = {1, 2, 3, 4, 5};
    int arr_sum = array_sum(arr, 5);
    if (arr_sum != 15) {
        printf("Array sum test failed: %d\n", arr_sum);
        return 1;
    }

    if (p_sum != 7.0) {
        printf("Struct test failed: %d\n", p_sum);
        return 1;
    }

    int value = 0;
    if (a < b) {
        value = 1;
    }
    else {
        value = 2;
    }
    if (value != 1) {
        printf("Branching test failed: %d\n", value);
        return 1;
    }

    printf("All tests passed successfully!\n");
    return 0;
}
