class ABCD
{
private:
    int x;
    int y = 8;
public:
    void setter(int a){
        x = a;
        return;
    }

    void setter(int a, int b){
        x = a;
        y = b;
        return;
    }

    int getx(){
        return x;
    }

    int gety(){
        return y;
    }
};

int main()
{
    ABCD a;
    a.setter(7);
    a.setter(7, 8);

    int p = a.getx();
    int q = a.gety();

    return 0;
}
