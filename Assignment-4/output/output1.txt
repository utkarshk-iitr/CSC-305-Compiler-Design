section .data
    str_1 db `%d\n`, 0
    str_2 db `%d\n`, 0

section .text
    global main
    extern printf
    extern scanf
    extern malloc
    extern free
    extern nullptr

ABCD.setter_int:
    push ebp
    mov ebp, esp
    sub esp, 0
    mov esp, ebp
    pop ebp
    ret
ABCD.setter_int_int:
    push ebp
    mov ebp, esp
    sub esp, 0
    mov eax, [ebp+8]
    mov [ecx+0], eax
    mov eax, [ebp+12]
    mov [ecx+4], eax
    mov esp, ebp
    pop ebp
    ret
ABCD.getx:
    push ebp
    mov ebp, esp
    sub esp, 0
    mov eax, [ecx+0]
    mov esp, ebp
    pop ebp
    ret
ABCD.gety:
    push ebp
    mov ebp, esp
    sub esp, 0
    mov eax, [ecx+4]
    mov esp, ebp
    pop ebp
    ret
main:
    push ebp
    mov ebp, esp
    sub esp, 32
    lea ecx, [ebp-8]
    push 7
    call ABCD.setter_int
    add esp, 4
    lea ecx, [ebp-8]
    push 8
    push 7
    call ABCD.setter_int_int
    add esp, 8
    call ABCD.getx
    mov [ebp-12], eax
    mov eax, [ebp-12]
    mov [ebp-16], eax
    call ABCD.gety
    mov [ebp-20], eax
    mov eax, [ebp-20]
    mov [ebp-24], eax
    mov eax, [ebp-16]
    push eax
    mov eax, str_1
    push eax
    call printf
    mov [ebp-28], eax
    add esp, 8
    mov eax, [ebp-24]
    push eax
    mov eax, str_2
    push eax
    call printf
    mov [ebp-32], eax
    add esp, 8
    xor eax, eax
    mov esp, ebp
    pop ebp
    ret
