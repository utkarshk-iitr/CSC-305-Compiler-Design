    
    incrementCallCount.callCount dd 0
    str_1 db "Inside incrementCallCount, callCount = %d\n", 0
    str_2 db "Static variable test:\n", 0
    main.localStatic dd 100
    str_3 db "Local static variable: %d\n", 0
    str_4 db "Modified local static: %d\n", 0
    
incrementCallCount:
    push ebp
    mov ebp, esp
    sub esp, 12
    [ebp - 8] = [incrementCallCount.callCount]
    [incrementCallCount.callCount] = [ebp - 8] + 1;
    sub esp, 4
    param [esp] , str_1
    sub esp, 4
    param [esp] , [incrementCallCount.callCount]
    [ebp - 12] = call printf, 2
    add esp, 8
    return
    
main:
    push ebp
    mov ebp, esp
    sub esp, 20
    sub esp, 4
    param [esp] , str_2
    [ebp - 4] = call printf, 1
    add esp, 4
    [ebp - 4] = call incrementCallCount, 0;
    [ebp - 4] = call incrementCallCount, 0;
    [ebp - 4] = call incrementCallCount, 0;
    sub esp, 4
    param [esp] , str_3
    sub esp, 4
    param [esp] , [main.localStatic]
    [ebp - 12] = call printf, 2
    add esp, 8
    [ebp - 16] = [main.localStatic] + 10
    [main.localStatic] = [ebp - 16]
    sub esp, 4
    param [esp] , str_4
    sub esp, 4
    param [esp] , [main.localStatic]
    [ebp - 20] = call printf, 2
    add esp, 8
    return 0
