#include <stdio.h>

struct Node{
    Node* ch[26] ;
    bool flag ;

    Node() {
        flag = false;

        for(int i=0 ;i<26 ;i++) ch[i] = nullptr;
    }
};

class Trie{
    private :
    Node *root;

    public :
    Trie() {
        root = new Node();
    }


    void insert(string& s) {
        Node* temp = root ;
        for(char c : s) {
            int pos = c - 'a';
            if(!temp->ch[pos]) {
                temp->ch[pos] = new Node();
            }

            temp = (*temp).ch[pos] ;
        }
    }
};
