#include <stdio.h>

class Employee{
    private : 
    int salary;

    public :
    double loan {0};

    protected :
    string name;
};

bool IsValid(int x , int y) {
    if(!x && !y) cout<<"Origin"<<"\n";
    if(x<0 || y<0 || x >10 || y< 10) return true;

    return false;
}

void IncrementLoan(Employee& emp) {
    emp.loan += 1;
}

int main() {
    if(IsValid(5,7)) cout<<"Ha Valid Hai"<<"\t"<<"\n";
    Employee Raja ;
    IncrementLoan(Raja);
    IncrementLoan(Raja);
    IncrementLoan(Raja);

    Employee* Rani = &Raja;
    Rani->salary++;
    Rani->salary--;

    for(int i=0 ;i<90 ;i++) {
        if(i > 40) continue;
        cout<<i<<" ";
    }
    cout<<"\n";

    if(Rani == NULL) cout<<"No Match Found"<<"\n";

    return 0;
}
